34 { factor_traits<T>::is_dimensional } -> std::convertible_to<bool>;
35
36 // If the factor does not have dimensionality, it must have a magnitude. This
37 // member is `true` if the magnitude of the base is representable by a ratio.
38 // Note this only considers if the *base* is rational; it ignores any
39 // irrationality caused by a fractional exponent.
40 { factor_traits<T>::is_rational_value } -> std::convertible_to<bool>;
41
42 // If the factor is not dimensional, and its base magnitude is representable
43 // by a ratio, this is the base's rational value.
44 { factor_traits<T>::rational_value } -> std::convertible_to<ratio>;
45
46 // If the factor is not dimensional, and its base magnitude is not
47 // representable by a ratio, this is the base's irrational value.
48 { factor_traits<T>::irrational_value } -> std::convertible_to<long double>;

51 // or dimensional factor).
