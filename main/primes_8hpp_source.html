58constexpr void try_prime_factor(std::vector<prime_factor> &out,
59 std::intmax_t &value, ratio exponent,
60 std::intmax_t candidate_prime) {
61 if (value % candidate_prime != 0) {
62 // Prime does not divide value.
63 return;
64 }
65
66 // Divide value by candidate prime as far as possible.
67 prime_factor current(candidate_prime);
68 do {
69 current.exponent.num++;
70 value /= candidate_prime;
71 } while (value % candidate_prime == 0);
72
73 // Push the prime factor. Recall the value being factored is already raised to
74 // an outer exponent.
75 current.exponent *= exponent;
76 out.push_back(current);
77}

89constexpr void prime_factorize_whole_number(std::vector<prime_factor> &out,
90 std::intmax_t &value,
91 ratio exponent) {
92
93 // Test primes in batches. Before each batch, check if the current prime is
94 // too high to consider checking further.
95 for (std::size_t i = 0; i < PRIMES_TABLE_SIZE; i += PRIMES_TABLE_BATCH_SIZE) {
96 if (PRIMES_TABLE[i] * PRIMES_TABLE[i] > value) {
97 // This batch contains primes too high to bother checking.
98 break;
99 }
100
101 // Test each prime in the batch.
102 const std::size_t batch_end = i + PRIMES_TABLE_BATCH_SIZE;
103 for (std::size_t j = 0; j < batch_end; ++j) {
104 try_prime_factor(out, value, exponent, PRIMES_TABLE[j]);
105 }
106 }
107
108 // If there is any remaining value, that means no primes in the table could
109 // divide this value. Let's **ASSUME** the remaining portion is a prime
110 // number.
111 if (value > 1) {
112 out.emplace_back(value, exponent);
113 }
114}

144constexpr void prime_factorize(std::vector<prime_factor> &out, ratio base,
145 ratio exponent) {
146 base.normalize_sign();
147 if (base.num < 0) {
148 // Push a factor of -1, then factorize the positive value instead.
149 out.push_back(prime_factor{-1, exponent});
150 base.num = -base.num;
151 }
152 prime_factorize_whole_number(out, base.num, exponent);
153 prime_factorize_whole_number(out, base.den, -exponent);
154}

160constexpr void combine_prime_factors(std::vector<prime_factor> &prime_factors) {
161 if (prime_factors.empty()) {
162 return;
163 }
164
165 // Combine factors with the same base.
166 std::sort(prime_factors.begin(), prime_factors.end());
167 auto it = prime_factors.begin();
168 auto cur = it;
169 ++it;
170 bool combine = false;
171 while (it != prime_factors.end()) {
172 if (it->base == cur->base) {
173 cur->exponent += it->exponent;
174 combine = true;
175 ++it;
176 continue;
177 }
178 if (combine) {
179 if (!cur->exponent.is_zero()) {
180 ++cur;
181 }
182 it = prime_factors.erase(cur, it);
183 combine = false;
184 }
185 cur = it;
186 ++it;
187 }
188 if (combine) {
189 if (!cur->exponent.is_zero()) {
190 ++cur;
191 }
192 prime_factors.erase(cur, it);
193 } else if (cur->exponent.is_zero()) {
194 prime_factors.erase(cur, it);
195 }
196}
