162template <rep ToRep, units ToUnits, units FromUnits, rep FromRep>
163requires quantity_losslessly_convertible_to<FromRep, FromUnits, ToRep, ToUnits>
164constexpr ToRep convert_losslessly(FromRep &&from_value) {
165 if constexpr (units_equivalent_to<FromUnits, ToUnits>) {
166 return rep_traits<ToRep>::lossless_cast(std::forward<FromRep>(from_value));
167
168 } else {
169 relaxed_scale_t<ToRep, units_conversion_t<FromUnits, ToUnits>> scale =
170 units_conversion_v<FromUnits, ToUnits>;
171 return rep_traits<ToRep>::lossless_cast(std::forward<FromRep>(from_value) *
172 scale);
173 }
174}
